#include <stdio.h>

long long countNumbers(int r) {
    long long dp[31][2];
    
    // Ініціалізація початкових значень
    dp[1][0] = 2; // Однорозрядні числа: 5 та 9
    dp[1][1] = 2; // Однорозрядні числа без поручних однакових цифр: 5 та 9
    
    // Обчислення кількості чисел
    for (int i = 2; i <= r; i++) {
        dp[i][0] = dp[i - 1][0] + dp[i - 1][1]; // Числа з поручними однаковими цифрами
        dp[i][1] = dp[i - 1][0] + dp[i - 1][1]; // Числа без поручних однакових цифр
    }
    
    return dp[r][0] + dp[r][1];
}

int main() {
    int r;

    printf("Введіть кількість розрядів р: ");
    scanf("%d", &r);

    long long numberCount = countNumbers(r);

    printf("Кількість чисел із %d розрядів: %lld\n", r, numberCount);

    return 0;
}
